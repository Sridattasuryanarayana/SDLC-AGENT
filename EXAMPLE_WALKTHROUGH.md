```python
# backend/calculator.py
# Generated by Developer Agent

class Calculator:
    """A simple calculator with error handling."""
    
    def add(self, a: float, b: float) -> float:
        """Add two numbers."""
        return a + b
    
    def subtract(self, a: float, b: float) -> float:
        """Subtract two numbers."""
        return a - b
    
    def multiply(self, a: float, b: float) -> float:
        """Multiply two numbers."""
        return a * b
    
    def divide(self, a: float, b: float) -> float:
        """Divide two numbers with error handling."""
        if b == 0:
            raise ValueError("Cannot divide by zero")
        return a / b
    
    def calculate(self, operation: str, a: float, b: float) -> float:
        """Execute calculation based on operation."""
        operations = {
            'add': self.add,
            'subtract': self.subtract,
            'multiply': self.multiply,
            'divide': self.divide
        }
        
        if operation not in operations:
            raise ValueError(f"Unknown operation: {operation}")
        
        return operations[operation](a, b)


# backend/routes.py
# Generated by Developer Agent

from fastapi import FastAPI, HTTPException
from pydantic import BaseModel
from calculator import Calculator

app = FastAPI(title="Calculator API")
calc = Calculator()

class CalculationRequest(BaseModel):
    operation: str
    num1: float
    num2: float

class CalculationResponse(BaseModel):
    result: float
    operation: str
    num1: float
    num2: float

@app.post("/calculate", response_model=CalculationResponse)
async def calculate(request: CalculationRequest):
    """Calculate based on operation and two numbers."""
    try:
        result = calc.calculate(request.operation, request.num1, request.num2)
        return CalculationResponse(
            result=result,
            operation=request.operation,
            num1=request.num1,
            num2=request.num2
        )
    except ValueError as e:
        raise HTTPException(status_code=400, detail=str(e))


# frontend/src/components/Calculator.tsx
// Generated by Developer Agent

import React, { useState } from 'react';

const Calculator: React.FC = () => {
  const [num1, setNum1] = useState<number>(0);
  const [num2, setNum2] = useState<number>(0);
  const [operation, setOperation] = useState<string>('add');
  const [result, setResult] = useState<number | null>(null);
  const [error, setError] = useState<string | null>(null);

  const handleCalculate = async () => {
    try {
      const response = await fetch('/calculate', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          operation,
          num1,
          num2
        })
      });

      if (!response.ok) {
        const data = await response.json();
        setError(data.detail);
        setResult(null);
        return;
      }

      const data = await response.json();
      setResult(data.result);
      setError(null);
    } catch (err) {
      setError('Failed to calculate');
      setResult(null);
    }
  };

  return (
    <div className="calculator">
      <h1>Simple Calculator</h1>
      
      <input 
        type="number" 
        value={num1} 
        onChange={(e) => setNum1(parseFloat(e.target.value))}
        placeholder="First number"
      />
      
      <select value={operation} onChange={(e) => setOperation(e.target.value)}>
        <option value="add">+</option>
        <option value="subtract">-</option>
        <option value="multiply">×</option>
        <option value="divide">÷</option>
      </select>
      
      <input 
        type="number" 
        value={num2} 
        onChange={(e) => setNum2(parseFloat(e.target.value))}
        placeholder="Second number"
      />
      
      <button onClick={handleCalculate}>Calculate</button>
      
      {error && <div className="error">{error}</div>}
      {result !== null && <div className="result">Result: {result}</div>}
    </div>
  );
};

export default Calculator;
```

```python
class TestCalculator:

        assert calc.add(0, 0) == 0

    def test_subtract(self, calc):
        assert calc.subtract(10, 3) == 7
        assert calc.subtract(-5, 3) == -8
    
    # Test Multiplication
    def test_multiply(self, calc):
        assert calc.multiply(5, 3) == 15
        assert calc.multiply(-5, 3) == -15
        assert calc.multiply(0, 100) == 0
    
    # Test Division
    def test_divide(self, calc):
        assert calc.divide(10, 2) == 5
        assert calc.divide(7, 2) == 3.5
    
    # Test Division by Zero (Error Handling)
    def test_divide_by_zero(self, calc):
        with pytest.raises(ValueError) as exc_info:
            calc.divide(10, 0)
        assert "Cannot divide by zero" in str(exc_info.value)
    
    # Test Invalid Operation
    def test_invalid_operation(self, calc):
        with pytest.raises(ValueError) as exc_info:
            calc.calculate('power', 2, 3)
        assert "Unknown operation" in str(exc_info.value)
```
